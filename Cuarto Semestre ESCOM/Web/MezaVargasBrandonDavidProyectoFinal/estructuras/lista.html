                typedef struct nodo{
                  int dato;
                  struct nodo *next;
                }Nodo;

                Nodo *crearNodo(int dato){
                  Nodo *nuevo = NULL;
                  nuevo = (Nodo*)malloc(sizeof(Nodo));
                  if( nuevo != NULL ){
                    nuevo-> dato = dato;
                    nuevo-> next = NULL;
                  }
                  return nuevo;
                }
